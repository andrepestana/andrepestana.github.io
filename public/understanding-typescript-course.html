    • Basic & Basic Types
      • number
        • All number (integers, float)
          • 1
          • 5.3
          • -10
      • string
        • All text values
          •  'Hi'
          • "Hi"
          • `Hi`
      • boolean
        • true or false only
          • true
          • false
      • object
        • Any JS object
          • { age: 30 }
        • Can be nested
          • {
  age: number;
  name: {
    firstName: string;
    lastName: string;
}
      • Array
        • can flexible
          • any[]
        • or strict
          • string[]
      • Tuple
        • An array with fixed types and length to have
      • Enum
        • Enumerated specific identifiers
          • can have number or string as values
          • enum Role { ADMIN, MANAGER }
      • any
        • Most flexible type
      • Union
        • const id: string | number;
      • Literal Types
        • const result: 'ok' | 'fail';
      • Type Aliases / Custom Types
        • type Combinable = string | number;
      • Function Return Types & "void"
        • function printResult(num: number): void {
  console.log('Result is: ' + num);
}
      • Functions as Types
        • let combineValues: () => number;
      • Function Types & Callbacks
        • function addHandler(n1: number, n2: number), cb: (num: number) => void) {
  const result = n1 + n2;
  cb(result);
}
      • unknown
        •     more restrictive than "any"
        • a "string" variable accepts receiving a "any" variable value but not an "unknown" variable value
      • never
        • return type to indicate it never returns a value
